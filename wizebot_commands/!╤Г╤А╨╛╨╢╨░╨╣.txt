// WizeBot command: !урожай
// После сбора урожая синкает сайт через MOOSE_SYNC longtext для nico_moose.

let now = Date.now();

let farmPlayers = [];

try {
  let raw = JS.wizebot.get_var("farm_players");
  farmPlayers = raw ? JSON.parse(raw) : [];
} catch (_) {
  JS.wizebot.send_chat_message("⚠️ Ошибка чтения фермеров.");
  JS.utils.stop();
}

if (!Array.isArray(farmPlayers) || farmPlayers.length === 0) {
  JS.wizebot.send_chat_message("📋 Нет активных ферм.");
  JS.utils.stop();
}

let plantTypes = JSON.parse(JS.wizebot.get_var("farm_data_plants") || "[]");
let animalTypes = JSON.parse(JS.wizebot.get_var("farm_data_animals") || "[]");
let buildingsConfig = JSON.parse(JS.wizebot.get_var("farm_data_buildings") || "{}");

function readVar(key, fallback) {
  let val = JS.wizebot.get_var(key);
  if (val === null || val === undefined || val === '') return fallback;
  return val;
}

function readJsonVar(key, fallback) {
  try {
    return JSON.parse(readVar(key, fallback));
  } catch (_) {
    try { return JSON.parse(fallback); } catch (__) { return null; }
  }
}

function readIntVar(key, fallback) {
  let n = parseInt(readVar(key, String(fallback)), 10);
  return Number.isFinite(n) ? n : fallback;
}

function readCurrency(user) {
  let n = parseInt(JS.wizebot.call_tag("currency", ["get", user]) || "0", 10);
  return Number.isFinite(n) ? n : 0;
}

function emitMooseSync(login) {
  let farm = readJsonVar('farm_' + login, '{}') || {};
  if (!farm || typeof farm !== 'object' || !Object.keys(farm).length) return false;

  let payload = {
    sync_version: 2,
    sync_source: 'wizebot_harvest_longtext',
    login: login,
    display_name: login,
    twitch_balance: readCurrency(login),
    farm: farm,
    farm_balance: readIntVar('farm_virtual_balance_' + login, 0),
    upgrade_balance: readIntVar('farm_upgrade_balance_' + login, 0),
    total_income: readIntVar('farm_total_income_' + login, 0),
    last_collect_at: readIntVar('farm_last_' + login, 0),
    license_level: readIntVar('farm_license_' + login, 0),
    protection_level: readIntVar('farm_protection_level_' + login, 0),
    raid_power: readIntVar('farm_raid_power_' + login, 0),
    turret: readJsonVar('farm_defense_building_' + login, '{}') || {},
    configs: {
      plants: readJsonVar('farm_data_plants', '[]') || [],
      animals: readJsonVar('farm_data_animals', '[]') || [],
      buildings: readJsonVar('farm_data_buildings', '{}') || {},
      licenses: readJsonVar('farm_data_licenses', '{}') || {},
      parts_required: readJsonVar('farm_data_parts_required', '{}') || {},
      turret_upgrades: readJsonVar('farm_data_turret_upgrades', '[]') || []
    },
    globals: {
      farm_parts_stock: readIntVar('farm_parts_stock', 20000),
      farm_parts_sold_total: readIntVar('farm_parts_sold_total', 0),
      farm_parts_bought_total: readIntVar('farm_parts_bought_total', 0)
    }
  };

  let json = JSON.stringify(payload, null, 2);
  let link = JS.wizebot.create_longtext('MOOSE FARM HARVEST SYNC ' + login, json);
  JS.wizebot.send_chat_message(`MOOSE_SYNC ${login} ${link}`);
  return true;
}

let reportData = [];

farmPlayers.forEach(player => {
  let raw = JS.wizebot.get_var("farm_" + player);
  if (!raw || raw === "null" || raw === "{}") return;

  let farm = {};
  try {
    farm = JSON.parse(raw);
  } catch (_) {
    return;
  }

  if (!farm || typeof farm !== "object" || !farm.level) return;

  farm.resources = farm.resources || {};
  farm.unlocked_at = farm.unlocked_at || {};
  farm.unlocked_at_ani = farm.unlocked_at_ani || {};
  farm.buildings = farm.buildings || {};

  let level = farm.level || 0;
  let last = parseInt(JS.wizebot.get_var("farm_last_" + player) || now, 10);
  let minutes = Math.min(60, Math.floor((now - last) / 60000));

  let saved = farm.savedPassive || 0;
  let passiveFromLevel = level * 2 * (minutes / 60);
  let buildingIncome = 0;

  for (let key in farm.buildings) {
    let lvl = farm.buildings[key] || 0;
    let conf = buildingsConfig[key];
    if (conf) {
      if (conf.coinsPerHour !== undefined) buildingIncome += conf.coinsPerHour * lvl * (minutes / 60);
      if (conf.coinsPerLevel !== undefined) buildingIncome += conf.coinsPerLevel * lvl;
    }
  }

  let totalPassive = passiveFromLevel + buildingIncome + saved;
  let maxPassive = (level * 2) + buildingIncome;
  if (totalPassive > maxPassive) {
    farm.savedPassive = totalPassive - maxPassive;
    totalPassive = maxPassive;
  } else {
    farm.savedPassive = 0;
  }

  let factoryLvl = farm.buildings["фабрика"] || 0;
  let plantLvl = farm.buildings["завод"] || 0;
  let mineLvl = farm.buildings["шахта"] || 0;

  let harvest = 0;
  let parts = 0;
  let weapons = 0;
  let shields = 0;

  plantTypes.forEach(p => {
    if (level >= p.level) {
      let key = p.key;
      let res = farm.resources[key] || 0;
      let unlockedAt = farm.unlocked_at[key] || now;
      let mins = Math.min(60, Math.floor((now - unlockedAt) / 60000));
      let growth = (p.base + p.perLevel * (level - 1)) * (mins / 60);
      let profit = (res + growth) * p.multiplier;
      harvest += profit;
      farm.resources[key] = 0;
      farm.unlocked_at[key] = now;
    }
  });

  animalTypes.forEach(a => {
    if (level >= a.level) {
      let key = a.key;
      let res = farm.resources[key] || 0;
      let unlockedAt = farm.unlocked_at_ani[key] || now;
      let mins = Math.min(60, Math.floor((now - unlockedAt) / 60000));
      let growth = (a.base + a.perLevel * (level - 1)) * (mins / 60);
      let profit = (res + growth) * a.multiplier;
      harvest += profit;
      farm.resources[key] = 0;
      farm.unlocked_at_ani[key] = now;
    }
  });

  if (plantLvl > 0) {
    let conf = buildingsConfig["завод"];
    if (conf) {
      let baseParts = conf.baseProduction + conf.perLevel * (plantLvl - 1);
      let factoryPercent = 0;
      if (factoryLvl > 0) {
        let factoryConf = buildingsConfig["фабрика"];
        if (factoryConf) factoryPercent = factoryConf.baseProduction + factoryConf.perLevel * (factoryLvl - 1);
      }
      let partsAfterFactory = baseParts * (1 + factoryPercent / 100);
      let minePercent = mineLvl;
      parts = Math.round(partsAfterFactory * (1 + minePercent / 100));
      farm.resources.parts = (farm.resources.parts || 0) + parts;
    }
  }

  if (farm.buildings["укрепления"]) {
    let lvl = farm.buildings["укрепления"];
    let conf = buildingsConfig["укрепления"];
    if (conf) {
      let produced = conf.baseProduction + (lvl - 1) * conf.perLevel;
      let current = farm.resources.shield || 0;
      let maxShield = produced * 2;
      let gained = Math.min(produced, Math.max(0, maxShield - current));
      if (gained > 0) {
        farm.resources.shield = current + gained;
        shields = gained;
      }
    }
  }

  if (farm.buildings["кузница"]) {
    let lvl = farm.buildings["кузница"];
    let conf = buildingsConfig["кузница"];
    if (conf) {
      let produced = conf.baseProduction + (lvl - 1) * conf.perLevel;
      let current = farm.resources.weapon || 0;
      let maxWeapon = lvl * 10;
      if (current < maxWeapon) {
        weapons = Math.min(produced, maxWeapon - current);
        farm.resources.weapon = current + weapons;
      }
    }
  }

  let zavodBonus = 0;
  let fabrikaBonus = 0;
  let mineBonus = 0;

  if (farm.buildings["завод"]) {
    let lvl = farm.buildings["завод"];
    if (lvl > 0) zavodBonus = lvl * 2000;
  }
  if (farm.buildings["фабрика"]) {
    let lvl = farm.buildings["фабрика"];
    if (lvl > 0) fabrikaBonus = lvl * 4000;
  }
  if (farm.buildings["шахта"]) {
    let lvl = farm.buildings["шахта"];
    if (lvl > 0) mineBonus = Math.floor((zavodBonus + fabrikaBonus) * (lvl / 100));
  }

  let upgradeAmountTotal = zavodBonus + fabrikaBonus + mineBonus;
  farm.zavodBonus = zavodBonus;
  farm.fabrikaBonus = fabrikaBonus;
  farm.mineBonus = mineBonus;

  if (upgradeAmountTotal > 0) {
    let upgradeKey = "farm_upgrade_balance_" + player;
    let currentUpgrade = parseInt(JS.wizebot.get_var(upgradeKey) || "0", 10);
    JS.wizebot.set_var(upgradeKey, currentUpgrade + upgradeAmountTotal);
  }

  let final = Math.round(totalPassive + harvest);
  let balanceKey = "farm_virtual_balance_" + player;
  let balance = parseInt(JS.wizebot.get_var(balanceKey) || "0", 10);
  balance += final;
  JS.wizebot.set_var(balanceKey, balance);

  let totalIncomeKey = "farm_total_income_" + player;
  let totalEarned = parseFloat(JS.wizebot.get_var(totalIncomeKey) || "0") + final;
  JS.wizebot.set_var(totalIncomeKey, totalEarned.toFixed(2));

  let protectionLevel = parseInt(JS.wizebot.get_var("farm_protection_level_" + player) || "0", 10);
  let displayPercent = protectionLevel > 0 ? (protectionLevel * 0.5).toFixed(1) : 0;
  let defenseText = protectionLevel > 0
    ? `🛡 Защита: уровень ${protectionLevel} (${displayPercent}%)`
    : "🛡 Защита не активирована";

  JS.wizebot.set_var("farm_" + player, JSON.stringify(farm));
  JS.wizebot.set_var("farm_last_" + player, now);

  reportData.push({
    player,
    passive: Math.round(totalPassive),
    harvest: Math.round(harvest),
    zavodBonus: farm.zavodBonus || 0,
    fabrikaBonus: farm.fabrikaBonus || 0,
    parts,
    weapons,
    shields,
    mineBonus: farm.mineBonus || 0,
    mine: farm.resources.mine || {},
    total: final,
    totalAll: Math.round(totalEarned),
    defenseText
  });
});

reportData.sort((a, b) => b.total - a.total);

let lines = reportData.map((e, i) => {
  let out = `${i + 1}) 👤 ${e.player}\n📥 Пассив: ${e.passive}💰 🌿 Урожай: ${e.harvest}💰 💵 Итог сбора: ${e.total}💰 (Всего: ${e.totalAll}💰)`;
  let info = [];
  if (e.zavodBonus > 0 || e.fabrikaBonus > 0 || e.mineBonus > 0) {
    const bonuses = [];
    if (e.zavodBonus > 0) bonuses.push(`🏭 ${e.zavodBonus}`);
    if (e.fabrikaBonus > 0) bonuses.push(`🏗 ${e.fabrikaBonus}`);
    if (e.mineBonus > 0) bonuses.push(`⛏ ${e.mineBonus}`);
    info.push(`💎 Бонусы: ${bonuses.join(" + ")}`);
  }
  if (e.weapons > 0) info.push(`⚔️ ${e.weapons}`);
  if (e.parts > 0) info.push(`🔧 ${Math.round(e.parts)}`);
  if (e.shields > 0) info.push(`🛡 ${e.shields}`);
  if (e.mine && typeof e.mine === 'object') {
    const emojis = { aluminium: '🪨', copper: '🪙', titanium: '🔩', ruby: '💎', uranium: '🧬' };
    for (let key in e.mine) if (e.mine[key] > 0) info.push(`${emojis[key] || '❔'} ${e.mine[key]}`);
  }
  if (info.length > 0) out += `\nПостройки — ${info.join(' | ')}`;
  if (e.defenseText) out += `\n${e.defenseText}`;
  return out;
});

let mooseSyncSent = emitMooseSync('nico_moose');

let full = '📦 Сбор урожая:\n\n' + lines.join('\n\n');
full += `\n\n🔄 MOOSE_SYNC для сайта: ${mooseSyncSent ? 'отправлен' : 'не отправлен'}`;

let link = JS.wizebot.create_longtext('Сбор урожая', full);
JS.wizebot.send_chat_message(`🌾 Урожай собран! ${mooseSyncSent ? 'Сайт обновляется.' : 'Sync не отправлен.'} Подробнее: ${link}`);
